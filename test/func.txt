    val Int a;

function add(Int x, Int y) returns Int:
    --printInt(x);
    val Int n;
    n = x + y;
    return n;
end

function sub(Int x, Int y) returns Int:
    return x - y;
end

function div(Int x, Int y) returns Int:
    return x / y;
end



function multi(Int x, Int y) returns Int:
    val Int l;
    l = x *y;
    return l;
end
